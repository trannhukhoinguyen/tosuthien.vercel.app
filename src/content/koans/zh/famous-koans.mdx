import FamousKoans from "../../../db/koans/vi/FamousKoans.tsx";
import VintagePost from "../../../layouts/components/VintagePost.astro";

<div class="container vi-koans">

  {
    FamousKoans
     ?.filter(koan => koan.id >= 1 && koan.id <= 10)
     ?.map(koan => {
        if (koan.id%10 === 1) {
          return <VintagePost
                        bookTitle={koan.bookTitle}
                        author={koan.author}
                        translator={koan.translator}
                        chapterTitle={koan.title}
                        image={koan.image}
                        main={koan.main}
                        hashTag={koan.hashTag}
          />
        } else {
          return <VintagePost
                       chapterTitle={koan.title}
                       image={koan.image}
                       main={koan.main}
                       hashTag={koan.hashTag}
          />
        }
      })
  }

</div>
