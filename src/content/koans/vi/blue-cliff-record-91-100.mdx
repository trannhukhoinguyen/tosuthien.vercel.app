import BlueCliffRecord from "../../../db/koans/vi/BlueCliffRecord.tsx";
import VintagePost from "../../../layouts/components/VintagePost.astro";

<div class="container vi-koans">

  {
    BlueCliffRecord
      ?.filter(koan => koan.id >= 91 && koan.id <= 100)
      ?.map(koan => {
        if (koan.id%10 === 1) {
          return <VintagePost
                    bookTitle={`${koan.bookTitle} (91 - 100)`}
                    author={koan.author}
                    translator={koan.translator}
                    chapterTitle={koan.title} image={koan.image}
                    intro={koan.intro} main={koan.main}
                    commentBeforeVerseExtra={koan.commentBeforeVerseExtra}
                    verseExtraOriginal={koan.verseExtraOriginal}
                    verseExtra={koan.verseExtra}
                    commentAfterVerseExtra={koan.commentAfterVerseExtra}
                    commentBeforeVerse={koan.commentBeforeVerse}
                    verseOriginal={koan.verseOriginal}
                    verse={koan.verse}
                    commentAfterVerse={koan.commentAfterVerse}
          />
        } else {
          return <VintagePost
                    chapterTitle={koan.title}
                    image={koan.image}
                    intro={koan.intro} main={koan.main}
                    commentBeforeVerseExtra={koan.commentBeforeVerseExtra}
                    verseExtraOriginal={koan.verseExtraOriginal}
                    verseExtra={koan.verseExtra}
                    commentAfterVerseExtra={koan.commentAfterVerseExtra}
                    commentBeforeVerse={koan.commentBeforeVerse}
                    verseOriginal={koan.verseOriginal}
                    verse={koan.verse}
                    commentAfterVerse={koan.commentAfterVerse}
          />
        }
      })
  }

</div>
